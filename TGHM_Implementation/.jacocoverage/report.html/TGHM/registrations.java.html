    public registrations(ManagementAuth mauth) {
        initComponents();
        ma = mauth;
        try
        {
            fs = Management.getRegistrants(ma);
            DefaultTableModel model = (DefaultTableModel) jTable1.getModel(); 
            for (int i = 0;i<fs.size();i++) {
                model.addRow(new Object[]{Integer.toString(i+1),fs.get(i).getName(),fs.get(i).getAuthCert().path,fs.get(i).getContact(),fs.get(i).getType()});
            }
            validate();
            repaint();
//            JOptionPane.showMessageDialog(null, "bf");
        } 
        catch (AuthenticationException ex) 
        {
            System.out.println("err:"+ex.getMessage());
        } 
        catch (UnableToConnectException ex) 
        {
            System.out.println("err:"+ex.getMessage());
        }
    }
    
    private void refresh()
    {
        
        try
        {
            fs = Management.getRegistrants(ma);
            DefaultTableModel model = (DefaultTableModel) jTable1.getModel();

                model.addRow(new Object[]{Integer.toString(i+1),fs.get(i).getName(),fs.get(i).getAuthCert().path,fs.get(i).getContact(),fs.get(i).getType()});

            validate();
            repaint();

        {
            System.out.println("err:"+ex.getMessage());
        } 
        catch (UnableToConnectException ex)

        }
    }

    private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButton1ActionPerformed
        try
        {
            if(acceptButton.isSelected())
            {
                Management.acceptRegistration(ma,fs.get(Integer.parseInt(jTextField1.getText())-1).getUsername());   
                refresh();
            }
            else if(rejectButton.isSelected())
            {
                Management.rejectRegistration(ma,fs.get(Integer.parseInt(jTextField1.getText())-1).getUsername());   
                refresh();
            }
            else
            {
                JOptionPane.showMessageDialog(null, "Please select an option");
            }
        } 
        catch (IllegalArgumentException ex) 
        {
            JOptionPane.showMessageDialog(null, "Invalid details");
        } 
        catch (UnableToConnectException ex) 
        {
            JOptionPane.showMessageDialog(null, "Unable to Connect!");
        } 
        catch (AuthenticationException ex) 
        {
            JOptionPane.showMessageDialog(null,ex.getMessage());
        }
        catch (Exception e)
        {
            JOptionPane.showMessageDialog(null, "Invalid sno");
        }
    }//GEN-LAST:event_jButton1ActionPerformed
