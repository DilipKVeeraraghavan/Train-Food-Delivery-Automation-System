    private ManagementAuth ma;

    public Agents(ManagementAuth mauth) {
        initComponents();
        ma = mauth;
        try 
        {
            /*agents list**/
            agents = Management.getAgents(ma);
            DefaultTableModel model = (DefaultTableModel)jTable1.getModel();
            model.setRowCount(0);
            for(int i = 0;i <agents.size();i++)
            {
                Agent agent  = agents.get(i);
                model.addRow(new Object[]{agent.getName(),agent.getSuccesfullDeliveries(),agent.getTotalDeliveries(),agent.getAgentFoodSupplier(),agent.getContact()});
            }
            
            validate();
            repaint();
        } 
        catch (AuthenticationException | UnableToConnectException | IllegalArgumentException ex) 
        {
            JOptionPane.showMessageDialog(null, ex.getMessage());
            ex.printStackTrace();
        }
    }
