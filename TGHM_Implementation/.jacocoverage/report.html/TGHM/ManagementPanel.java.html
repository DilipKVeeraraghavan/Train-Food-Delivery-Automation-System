    private void jButton6ActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButton6ActionPerformed
//        JOptionPane.showMessageDialog(null, "af");
        setInvisible();
        jPanel3.setVisible(true);
        try
        {
            fs = Management.getRegistrants(ma);
            ArrayList<String[]> data = new ArrayList<String[]>();
            String[] column          = {"Name","Certificate","Contact","Type"};
            for(int i=0;i<fs.size();i++)
            {
                String[] row = new String[4];
                row[0]       = fs.get(i).getName();
                row[1]       = fs.get(i).getAuthCert().path;
                row[2]       = fs.get(i).getContact();
                row[3]       = fs.get(i).getType();
                data.add(row);
            }
            String[][] ab = new String[fs.size()][4];
            data.toArray(ab);
            table = new JTable(ab,column);
            table.setSize(600,200);
            table.setLocation(30,30);
            jPanel3.add(table);
            table.setVisible(true);
            validate();
            repaint();
            jTextField1.setText("");
//            JOptionPane.showMessageDialog(null, "bf");
        } 
        catch (AuthenticationException ex) 
        {
            Logger.getLogger(ManagementPanel.class.getName()).log(Level.SEVERE, null, ex);
        } 
        catch (UnableToConnectException ex) 
        {
            Logger.getLogger(ManagementPanel.class.getName()).log(Level.SEVERE, null, ex);
        }
    }//GEN-LAST:event_jButton6ActionPerformed
