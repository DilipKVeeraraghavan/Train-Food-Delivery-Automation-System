    private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButton1ActionPerformed
        try
        {
            JFileChooser fc = new JFileChooser();
            FileNameExtensionFilter filter = new FileNameExtensionFilter(
            "Images", "jpg", "gif","bmp","png","jpeg");
            fc.setFileFilter(filter);
            fc.setCurrentDirectory(new java.io.File("."));
            fc.setDialogTitle("Choose Picture");
            fc.showOpenDialog(this);
            certField.setText(fc.getSelectedFile().getAbsolutePath());
            BufferedImage img = null;
            img = ImageIO.read(new File(fc.getSelectedFile().getAbsolutePath()));
            BufferedImage dimg = convertToBufferedImage( (img.getScaledInstance(label.getWidth()-3, label.getHeight()-3,Image.SCALE_SMOOTH) ));
            ImageIcon imageIcon = new ImageIcon(dimg);
            label.setIcon(imageIcon);
        }
        catch(Exception e)
        {
            JOptionPane.showMessageDialog(null, "Unable to upload file");
            certField.setText("");
        }
    }//GEN-LAST:event_jButton1ActionPerformed

    private void jToggleButton1ActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jToggleButton1ActionPerformed
        boolean notValid = false;
        notValid = (usernameField.getText().equals(""))||(passwordField.getText().equals(""))||(nameField.getText().equals(""))||(certField.getText().equals(""))||(contactField.getText().equals(""))||(agentNameField.getText().equals(""))||(agentContactField.getText().equals(""))||(agentUsernameField.getText().equals(""))||(agentPasswordField.getText().equals(""));
        notValid = notValid||!(foodChain.isSelected()||restaurant.isSelected());
        if(notValid == true)
        {
            JOptionPane.showMessageDialog(null, "Please check all fields");
        }
        else
        {
            try
            {
               File inputFile  = new File(certField.getText());
               File outputFile = new File("./images/"+inputFile.getName());
               Files.copy(inputFile.toPath(),outputFile.toPath(),StandardCopyOption.REPLACE_EXISTING);
               String type = "FoodChain";
               if(restaurant.isSelected())
               {
                   type = "Restaurant";
               }
               Management.sendRegistrationRequest(usernameField.getText(), passwordField.getText(), nameField.getText(),new Cert(certField.getText()),type , contactField.getText(), agentNameField.getText(), agentContactField.getText(), agentUsernameField.getText(), agentContactField.getText()); 
               JOptionPane.showMessageDialog(null, "done");
               this.setVisible(false);
            }
            catch(Exception e)
            {
                  JOptionPane.showMessageDialog(null, e.getMessage()); 
            }
        }
    }//GEN-LAST:event_jToggleButton1ActionPerformed

    public static BufferedImage convertToBufferedImage(Image image)
    {
        BufferedImage newImage = new BufferedImage(
        image.getWidth(null), image.getHeight(null),
        BufferedImage.TYPE_INT_ARGB);
        Graphics2D g = newImage.createGraphics();
        g.drawImage(image, 0, 0, null);
        g.dispose();
        return newImage;
    }
